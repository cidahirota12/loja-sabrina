⚙️ Configurar Sincronização com Google Sheets

Configure UMA VEZ. Após isso, todos os dispositivos sincronizam automaticamente.
📖 Como configurar — passo a passo
Acesse sheets.google.com e crie uma planilha nova. Dê o nome Loja Sabrina — Fornecedores.
Dentro da planilha, clique em Extensões → Apps Script.
Apague o código que aparecer e cole o código abaixo no lugar:
// Cole este código no Google Apps Script const SHEET_NAME_MATRIZ = 'Matriz'; const SHEET_NAME_HISTORICO = 'Historico'; function doGet(e) { const action = e.parameter.action || 'getData'; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'getData') { const mat = getSheetData(ss, SHEET_NAME_MATRIZ); const his = getSheetData(ss, SHEET_NAME_HISTORICO); return jsonResponse({ matriz: mat, historico: his }); } return jsonResponse({ error: 'Ação desconhecida' }); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (action === 'saveMatriz') { saveSheetData(ss, SHEET_NAME_MATRIZ, body.data, getMatrizHeaders()); return jsonResponse({ ok: true }); } if (action === 'saveHistorico') { saveSheetData(ss, SHEET_NAME_HISTORICO, body.data, getHistoricoHeaders()); return jsonResponse({ ok: true }); } if (action === 'saveAll') { saveSheetData(ss, SHEET_NAME_MATRIZ, body.matriz, getMatrizHeaders()); saveSheetData(ss, SHEET_NAME_HISTORICO, body.historico, getHistoricoHeaders()); return jsonResponse({ ok: true }); } return jsonResponse({ error: 'Ação inválida' }); } catch(err) { return jsonResponse({ error: err.message }); } } function getMatrizHeaders() { return ['fornecedor','cnpj','produto','un','qtde','precoNF','ipi', 'temST','vICMSST','pICMSST','stPorUnidade','custoUnit', 'reajuste','novoPreco','variacao','dataNF','nNF','status', 'markup','precoVenda','csosn','timestamp']; } function getHistoricoHeaders() { return ['data','hora','qtdeItens','fornecs','nfs']; } function getSheetData(ss, name) { let sheet = ss.getSheetByName(name); if (!sheet) return []; const vals = sheet.getDataRange().getValues(); if (vals.length < 2) return []; const headers = vals[0]; return vals.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i]; }); return obj; }); } function saveSheetData(ss, name, data, headers) { let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); sheet.clearContents(); if (!data || !data.length) { sheet.getRange(1,1,1,headers.length).setValues([headers]); return; } const rows = data.map(obj => headers.map(h => obj[h] !== undefined ? obj[h] : '')); sheet.getRange(1,1,1,headers.length).setValues([headers]); sheet.getRange(2,1,rows.length,headers.length).setValues(rows); } function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
